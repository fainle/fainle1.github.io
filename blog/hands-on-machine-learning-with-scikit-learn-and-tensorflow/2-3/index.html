    加载数据

    import pandas as pd
import os

DATA_ROOT_PATH = "../data/housing/"
FILES_NAME = 'housing.csv'

def load_housing_data(housing_PATH = DATA_ROOT_PATH):
    csv_path = os.path.join(housing_PATH, FILES_NAME)
    return pd.read_csv(csv_path)

    其中housing.csv 来至于 https://raw.githubusercontent.com/ageron/handson-ml/master/datasets/housing/housing.tgz

    housing = load_housing_data()
